git clone https://github.com/coolq4s/server-lnx.git && mv server-lnx/other/troubleshoot/varlog-troubleshoot.sh varlog-troubleshoot.sh && chmod +x varlog-troubleshoot.sh && ./varlog-troubleshoot.sh
```
